scene.onOverlapTile(SpriteKind.Player, assets.tile`food`, function (sprite, location) {
    music.baDing.play()
    tiles.setTileAt(location, assets.tile`grid`)
    info.changeScoreBy(1)
})

controller.A.onEvent(ControllerButtonEvent.Released, function () {
    startCountdown = 1
})

let mySprite: Sprite = null
let startCountdown = 0
let stillWalking = 5
scene.setBackgroundColor(1)
tiles.setTilemap(tilemap`level1`)
mySprite = sprites.create(assets.image`dino`, SpriteKind.Player)
tiles.placeOnTile(mySprite, tiles.getTileLocation(2, 4))
mySprite.x += -1
characterAnimations.loopFrames(
    mySprite,
    assets.animation`walk right`,
    500,
    characterAnimations.rule(Predicate.MovingLeft)
)
characterAnimations.loopFrames(
    mySprite,
    assets.animation`walk right`,
    500,
    characterAnimations.rule(Predicate.MovingRight)
)
game.onUpdateInterval(500, function () {
    if (info.score() >= 2) {
        game.over(true)
    } else if (startCountdown == 1) {
        stillWalking += -1
    }
    if (stillWalking < 0) {
        game.over(false)
    }
})

namespace sprites {
   
    //% block="step -->"
    export function step_right () {
        for (let index = 0; index < 8; index++) {
            mySprite.x += 2
            pause(50)
        }
        pause(500)
    }

    //% block="step <--"
    export function step_left () {
        for (let index = 0; index < 8; index++) {
            mySprite.x -= 2
            pause(50)
        }
        pause(500)
    }
}
