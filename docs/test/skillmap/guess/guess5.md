        mySprite.say("Yes! It was " + yourGuess + "!")
        pause(3000)
        game.over(true)
    } else {
        if (yourGuess < myNumber) {
            mySprite.say("Too low!")
        } else {
            mySprite.say("Too high!")
        }
        pause(3000)
        info.changeLifeBy(-1)
    }
}
